• cd -- смена директории
  • git clone "url" -- клонирование уже созданного репозитория
  • git stastus -- проверка статуса, наличие изменений и проверка в какой ветке мы сейчас
  • git branch "имя новой ветки" -- создание новой ветки
  • git checkout "имя ветки" -- переключение на другую (уже созданную ветку)
  • git add "file name" -- обавление файла в список "законтроленных"
  • git commit -- закомитить все добавленные под контроль файлы (при этом откроется текстовый редактор для создания заметки по коммиту)
  • git commit -m "message" -- закомитить с заметкой прямо из консоли, без текстового редакора
  • git log -- лог коммита
  • git push -- запушить в удаленный репозиторий тот коммит, который законтролен
  • git push -u origin "branch name" -- запушить веточку в удаленный репозиторий
  • git pull -- обновить локаьный репозиторий до состояния удаленного (sync)
